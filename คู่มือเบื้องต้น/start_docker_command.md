## ปิดทั้งระบบ:
-------------------------------------------
docker compose --profile forecast-sync down
-------------------------------------------


## เปิดทั้งระบบ:
-------------------------------------------
docker compose --profile forecast-sync up -d
-------------------------------------------


## เปิดทั้งระบบพร้อม rebuild:
-------------------------------------------
docker compose --profile forecast-sync up -d --build
-------------------------------------------


## คำสั่งเช็คว่า forecast_sync ทำงานอยู่รึเปล่า
-------------------------------------------
docker compose --profile forecast-sync logs --since 5m forecast-sync
-------------------------------------------
